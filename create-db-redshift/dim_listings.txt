CREATE TABLE public.dim_listings (
    listing_id BIGINT PRIMARY KEY,
    listing_name VARCHAR(1000),
    host_id BIGINT REFERENCES public.dim_hosts(host_id), 
    last_scraped_date DATE,
    neighbourhood VARCHAR(255),
    latitude DECIMAL(10, 7),
    longitude DECIMAL(10, 7),
    property_type VARCHAR(255),
    room_type VARCHAR(255),
    accommodates INT,
    bathrooms_text VARCHAR(255),
    bedrooms INT,
    beds INT,
    default_price DECIMAL(10, 2),
    default_min_nights INT,
    default_max_nights INT,
    has_availability BOOLEAN,
    instant_bookable BOOLEAN,
    number_of_reviews INT,
    number_of_reviews_ltm INT,
    first_review_date DATE,
    last_review_date DATE,
    review_score_rating DECIMAL(3, 2),
    review_score_cleanliness DECIMAL(3, 2),
    review_score_location DECIMAL(3, 2),
    reviews_per_month DECIMAL(5, 2)
);
